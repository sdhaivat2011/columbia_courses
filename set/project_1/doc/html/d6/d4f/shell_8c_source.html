01684 /* Read a single field of CSV text.  Compatible with rfc4180 and extended
01685 ** with the option of having a separator other than ",".
01686 **
01687 **   +  Input comes from p->in.
01688 **   +  Store results in p->z of length p->n.  Space to hold p->z comes
01689 **      from sqlite3_malloc().
01690 **   +  Use p->cSep as the separator.  The default is ",".
01691 **   +  Keep track of the line number in p->nLine.
01692 **   +  Store the character that terminates the field in p->cTerm.  Store
01693 **      EOF on end-of-file.
01694 **   +  Report syntax errors on stderr
01695 */
01696 static char *csv_read_one_field(CSVReader *p){
01697   int c, pc;
01698   int cSep = p->cSeparator;
01699   p->n = 0;
01700   c = fgetc(p->in);
01701   if( c==EOF || seenInterrupt ){
01702     p->cTerm = EOF;
01703     return 0;
01704   }
01705   if( c=='"' ){
01706     int startLine = p->nLine;
01707     int cQuote = c;
01708     pc = 0;
01709     while( 1 ){
01710       c = fgetc(p->in);
01711       if( c=='\n' ) p->nLine++;
01712       if( c==cQuote ){
01713         if( pc==cQuote ){
01714           pc = 0;
01715           continue;
01716         }
01717       }
01718       if( (c==cSep && pc==cQuote)
01719        || (c=='\n' && pc==cQuote)
01720        || (c=='\n' && pc=='\r' && p->n>=2 && p->z[p->n-2]==cQuote)
01721        || (c==EOF && pc==cQuote)
01722       ){
01723         do{ p->n--; }while( p->z[p->n]!=cQuote );
01724         p->cTerm = c;
01725         break;
01726       }
01727       if( pc==cQuote && c!='\r' ){
01728         fprintf(stderr, "%s:%d: unescaped %c character\n",
01729                 p->zFile, p->nLine, cQuote);
01730       }
01731       if( c==EOF ){
01732         fprintf(stderr, "%s:%d: unterminated %c-quoted field\n",
01733                 p->zFile, startLine, cQuote);
01734         p->cTerm = EOF;
01735         break;
01736       }
01737       csv_append_char(p, c);
01738       pc = c;
01739     }
01740   }else{
01741     while( c!=EOF && c!=cSep && c!='\n' ){
01742       csv_append_char(p, c);
01743       c = fgetc(p->in);
01744     }
01745     if( c=='\n' ){
01746       p->nLine++;

02843 /*
02844 ** Read input from *in and process it.  If *in==0 then input
02845 ** is interactive - the user is typing it it.  Otherwise, input
02846 ** is coming from a file or device.  A prompt is issued and history
02847 ** is saved only if input is interactive.  An interrupt signal will
02848 ** cause this routine to exit immediately, unless input is interactive.
02849 **
02850 ** Return the number of errors.
02851 */

02853   char *zLine = 0;          /* A single input line */

02855   int nLine;                /* Length of current line */
02856   int nSql = 0;             /* Bytes of zSql[] used */
02857   int nAlloc = 0;           /* Allocated zSql[] space */
02858   int nSqlPrior = 0;        /* Bytes of zSql[] used by prior line */
02859   char *zErrMsg;            /* Error message returned */
02860   int rc;                   /* Error code */
02861   int errCnt = 0;           /* Number of errors seen */
02862   int lineno = 0;           /* Current line number */
02863   int startline = 0;        /* Line number for start of current input */
02864 
02865   while( errCnt==0 || !bail_on_error || (in==0 && stdin_is_interactive) ){
02866     fflush(p->out);
02867     zLine = one_input_line(in, zLine, nSql>0);
02868     if( zLine==0 ){
02869       /* End of input */
02870       if( stdin_is_interactive ) printf("\n");
02871       break;
02872     }
02873     if( seenInterrupt ){
02874       if( in!=0 ) break;
02875       seenInterrupt = 0;
02876     }
02877     lineno++;
02878     if( nSql==0 && _all_whitespace(zLine) ) continue;
02879     if( zLine && zLine[0]=='.' && nSql==0 ){
02880       if( p->echoOn ) printf("%s\n", zLine);
02881       rc = do_meta_command(zLine, p);
02882       if( rc==2 ){ /* exit requested */
02883         break;
02884       }else if( rc ){
02885         errCnt++;
02886       }
02887       continue;
02888     }
02889     if( line_is_command_terminator(zLine) && line_is_complete(zSql, nSql) ){
02890       memcpy(zLine,";",2);
02891     }
02892     nLine = strlen30(zLine);
02893     if( nSql+nLine+2>=nAlloc ){
02894       nAlloc = nSql+nLine+100;
02895       zSql = realloc(zSql, nAlloc);
02896       if( zSql==0 ){
02897         fprintf(stderr, "Error: out of memory\n");
02898         exit(1);
02899       }
02900     }
02901     nSqlPrior = nSql;
02902     if( nSql==0 ){
02903       int i;
02904       for(i=0; zLine[i] && IsSpace(zLine[i]); i++){}
02905       assert( nAlloc>0 && zSql!=0 );
02906       memcpy(zSql, zLine+i, nLine+1-i);
02907       startline = lineno;
02908       nSql = nLine-i;
02909     }else{
02910       zSql[nSql++] = '\n';
02911       memcpy(zSql+nSql, zLine, nLine+1);
02912       nSql += nLine;
02913     }
02914     if( nSql && line_contains_semicolon(&zSql[nSqlPrior], nSql-nSqlPrior)
02915                 && sqlite3_complete(zSql) ){
02916       p->cnt = 0;
02917       open_db(p);
02918       BEGIN_TIMER;
02919       rc = shell_exec(p->db, zSql, shell_callback, p, &zErrMsg);
02920       END_TIMER;
02921       if( rc || zErrMsg ){
02922         char zPrefix[100];
02923         if( in!=0 || !stdin_is_interactive ){
02924           sqlite3_snprintf(sizeof(zPrefix), zPrefix, 
02925                            "Error: near line %d:", startline);
02926         }else{
02927           sqlite3_snprintf(sizeof(zPrefix), zPrefix, "Error:");
02928         }
02929         if( zErrMsg!=0 ){
02930           fprintf(stderr, "%s %s\n", zPrefix, zErrMsg);
02931           sqlite3_free(zErrMsg);
02932           zErrMsg = 0;
02933         }else{
02934           fprintf(stderr, "%s %s\n", zPrefix, sqlite3_errmsg(p->db));
02935         }
02936         errCnt++;
02937       }
02938       nSql = 0;
02939     }else if( nSql && _all_whitespace(zSql) ){
02940       nSql = 0;
02941     }
02942   }
02943   if( nSql ){
02944     if( !_all_whitespace(zSql) ){
02945       fprintf(stderr, "Error: incomplete SQL: %s\n", zSql);
02946     }
02947     free(zSql);
02948   }
02949   free(zLine);
02950   return errCnt>0;
