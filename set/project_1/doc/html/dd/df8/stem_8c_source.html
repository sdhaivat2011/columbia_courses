00099 /* cons(z, i) is TRUE <=> b[i] is a consonant. ('b' means 'z->b', but here
00100    and below we drop 'z->' in comments.
00101 */

00103 static int cons(struct stemmer * z, int i)
00104 {  switch (z->b[i])
00105    {  case 'a': case 'e': case 'i': case 'o': case 'u': return FALSE;
00106       case 'y': return (i == 0) ? TRUE : !cons(z, i - 1);
00107       default: return TRUE;
00108    }
00109 }

00111 /* m(z) measures the number of consonant sequences between 0 and j. if c is
00112    a consonant sequence and v a vowel sequence, and <..> indicates arbitrary
00113    presence,
00114 
00115       <c><v>       gives 0
00116       <c>vc<v>     gives 1
00117       <c>vcvc<v>   gives 2
00118       <c>vcvcvc<v> gives 3
00119       ....
00120 */
00121 
00122 static int m(struct stemmer * z)
00123 {  int n = 0;
00124    int i = 0;
00125    int j = z->j;
00126    while(TRUE)
00127    {  if (i > j) return n;
00128       if (! cons(z, i)) break; i++;
00129    }
00130    i++;
00131    while(TRUE)
00132    {  while(TRUE)
00133       {  if (i > j) return n;
00134             if (cons(z, i)) break;
00135             i++;
00136       }
00137       i++;
00138       n++;
00139       while(TRUE)
00140       {  if (i > j) return n;
00141          if (! cons(z, i)) break;
00142          i++;
00143       }
00144       i++;
00145    }
00146 }

00160 {
00161    char * b = z->b;

00167 /* cvc(z, i) is TRUE <=> i-2,i-1,i has the form consonant - vowel - consonant
00168    and also if the second c is not w,x or y. this is used when trying to
00169    restore an e at the end of a short word. e.g.
00170 
00171       cav(e), lov(e), hop(e), crim(e), but
00172       snow, box, tray.
00173 
00174 */
00175 
00176 static int cvc(struct stemmer * z, int i)
00177 {  if (i < 2 || !cons(z, i) || cons(z, i - 1) || !cons(z, i - 2)) return FALSE;
00178    {  int ch = z->b[i];
00179       if (ch  == 'w' || ch == 'x' || ch == 'y') return FALSE;
00180    }
00181    return TRUE;
00182 }

00184 /* ends(z, s) is TRUE <=> 0,...k ends with the string s. */
00185 
00186 static int ends(struct stemmer * z, char * s)
00187 {  int length = s[0];
00188    char * b = z->b;
00189    int k = z->k;
00190    if (s[length] != b[k]) return FALSE; /* tiny speed-up */
00191    if (length > k + 1) return FALSE;
00192    if (memcmp(b + k - length + 1, s + 1, length) != 0) return FALSE;
00193    z->j = k-length;
00194    return TRUE;
00195 }

00233 static void step1ab(struct stemmer * z)
00234 {
00235    char * b = z->b;
00236    if (b[z->k] == 's')
00237    {  if (ends(z, "\04" "sses")) z->k -= 2; else
00238       if (ends(z, "\03" "ies")) setto(z, "\01" "i"); else
00239       if (b[z->k - 1] != 's') z->k--;
00240    }
00241    if (ends(z, "\03" "eed")) { if (m(z) > 0) z->k--; } else
00242    if ((ends(z, "\02" "ed") || ends(z, "\03" "ing")) && vowelinstem(z))
00243    {  z->k = z->j;
00244       if (ends(z, "\02" "at")) setto(z, "\03" "ate"); else
00245       if (ends(z, "\02" "bl")) setto(z, "\03" "ble"); else
00246       if (ends(z, "\02" "iz")) setto(z, "\03" "ize"); else
00247       if (doublec(z, z->k))
00248       {  z->k--;
00249          {  int ch = b[z->k];
00250             if (ch == 'l' || ch == 's' || ch == 'z') z->k++;
00251          }
00252       }
